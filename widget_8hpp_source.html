419 template <typename... Attrs>
420 void add_attributes(Widget& w, Attrs&&... attrs) {
421  for (Attribute a : {attrs...}) {
422  if (!w.brush.has_attribute(a)) {

430 template <typename... Attrs>
431 void remove_attributes(Widget& w, Attrs&&... attrs) {
432  bool repaint{false};
433  for (const auto& a : {attrs...}) {
434  if (w.brush.has_attribute(a)) {
435  w.brush.remove_attribute(a);
436  repaint = true;
437  }
438  }
439  if (repaint) {
440  w.update();
441  }
