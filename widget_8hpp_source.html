200 // Breadth First Search
201 template <typename T, typename U>
202 auto Widget::find_child_impl(U u, const std::string& name)
203  -> decltype(u->template find_child<T>(name)) {
204  std::queue<U> search_queue;
205  search_queue.push(u);
206  while (!search_queue.empty()) {
207  auto current = search_queue.front();
208  search_queue.pop();
209  auto widg_T_ptr = dynamic_cast<T*>(current);
210  if (widg_T_ptr != nullptr && current->name() == name) {
211  return widg_T_ptr;
212  }
213  auto children = current->children();
214  std::for_each(std::begin(children), std::end(children),
215  [&search_queue](Widget* c) { search_queue.push(c); });
216  }
217  return nullptr;
218 }
