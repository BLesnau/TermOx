143  static constexpr auto hsl_to_rgb(HSL v) -> RGB
144  {
145  double const lightness = v.lightness / 100.;
146  double const saturation = v.saturation / 100.;
147 
148  auto const c = (1 - abs((2 * lightness) - 1.)) * saturation;
149  double const h_prime = v.hue / 60.;
150  double const x = c * (1. - abs(fmod(h_prime, 2.) - 1.));
151  double const m = lightness - (c / 2.);
152 
153  auto const c_ = static_cast<RGB::Value_t>((c + m) * 255);
154  auto const x_ = static_cast<RGB::Value_t>((x + m) * 255);
155  auto const m_ = static_cast<RGB::Value_t>(m * 255);
156 
157  if (v.hue < 60.)
158  return {c_, x_, m_};
159  if (v.hue < 120.)
160  return {x_, c_, m_};
161  if (v.hue < 180.)
162  return {m_, c_, x_};
163  if (v.hue < 240.)
164  return {m_, x_, c_};
165  if (v.hue < 300.)
166  return {x_, m_, c_};
167  if (v.hue < 360.)
168  return {c_, m_, x_};
169  else
170  return {0, 0, 0};
171  }

174  static /* constexpr */ auto rgb_to_hsl(RGB x) -> HSL
175  {
176  double const r_prime = x.red / 255.;
177  double const g_prime = x.green / 255.;
178  double const b_prime = x.blue / 255.;
179 
180  double const c_max = std::max({r_prime, g_prime, b_prime});
181  double const c_min = std::min({r_prime, g_prime, b_prime});
182  double const delta = c_max - c_min;
183 
184  double const lightness = (c_max + c_min) / 2.;
185  double const saturation = [&] {
186  if (delta == 0.)
187  return 0.;
188  double const den = 1. - abs(2. * lightness - 1.);
189  return delta / den;
190  }();
191  std::uint16_t const hue = [&] {
192  if (delta == 0.)
193  return 0.;
194  if (c_max == r_prime)
195  return 60. * fmod((g_prime - b_prime) / delta, 6.);
196  if (c_max == g_prime)
197  return 60. * (((b_prime - r_prime) / delta) + 2.);
198  if (c_max == b_prime)
199  return 60. * (((r_prime - g_prime) / delta) + 4.);
200  return .0;
201  }();
202  return {static_cast<std::uint16_t>(hue),
203  static_cast<std::uint8_t>(saturation * 100),
204  static_cast<std::uint8_t>(lightness * 100)};
205  }
