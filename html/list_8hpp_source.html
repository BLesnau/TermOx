81  p.move_cursor_on_put = true;
82  for (const auto& prop : properties_) {
83  p.put(Glyph_string{prop.name}.append(" | "));
84  }
85  std::size_t count{1};
86  for (const auto& item : items_) {
87  Glyph_string display;
88  for (const auto& prop : properties_) {
89  display.append(prop.get_value(item));

92  if (count == selected_index_) {
93  display.add_attributes(Attribute::Bold);
