191  auto find_offset(std::size_t text_length,
192  std::size_t box_length,
193  std::size_t extra_left,
194  std::size_t extra_right) -> std::size_t
195  {
196  switch (alignment_) {
197  case Align::Left:
198  case Align::Top: return left_top_offset();
199 
200  case Align::Center:
201  return apply_origin_offset(
202  center_offset(text_length,
203  box_length + extra_left + extra_right),
204  extra_left);
205 
206  case Align::Right:
207  case Align::Bottom:
208  return right_bottom_offset(text_length, box_length);
209  }
210  return 0uL;
211  }
