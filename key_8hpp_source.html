252 inline auto to_wchar(Key k) -> wchar_t
253 {
254  auto constexpr printable_low = wchar_t{32};
255  auto constexpr printable_high = wchar_t{126};
256  auto const value = static_cast<wchar_t>(k);
257  if (value < printable_low || value > printable_high)
258  return L'\0';
259  else
260  return static_cast<wchar_t>(value);
261 }
