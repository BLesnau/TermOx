141  // Update horizontal and vertical intervals used for bitmap calcs.
142  /* Called by resize_event and boundary change. This _will_ be incorrect for
143  * border changes, but that should be rare, since border updates from resize

146  {
147  auto const domain_w = distance(boundary_.west, boundary_.east);
148  interval_w_ = domain_w / this->width();
149  auto const domain_h = distance(boundary_.north, boundary_.south);
150  interval_h_ = domain_h / this->height();

154  void initialize_bitmap(Coordinates const& c)
155  {
156  auto const visual_width = this->width();
157  auto const visual_height = this->height();
158  if (visual_width == 0 || visual_height == 0)
159  return;
160 
161  auto const horizontal_offset =
162  interval_w_ == 0 ? 0 : distance(boundary_.west, c.x) / interval_w_;
163 
164  auto const distance_from_bottom =
165  interval_h_ == 0 ? 0 : distance(boundary_.north, c.y) / interval_h_;
166  auto const vertical_offset = visual_height - distance_from_bottom;
167 
168  auto& bitmap = map_[{static_cast<std::size_t>(horizontal_offset),
169  static_cast<std::size_t>(vertical_offset)}];
170 
171  auto const h_dec = std::fmod(horizontal_offset, 1.);
172  auto const v_dec = std::fmod(vertical_offset, 1.);
173  if (h_dec < 0.5) {
174  if (v_dec < 0.25)
175  bitmap.set_top_left();
176  else if (v_dec < 0.5)
177  bitmap.set_mid_top_left();
178  else if (v_dec < 0.75)
179  bitmap.set_mid_bottom_left();
180  else
181  bitmap.set_bottom_left();
182  }
183  else {
184  if (v_dec < 0.25)
185  bitmap.set_top_right();
186  else if (v_dec < 0.5)
187  bitmap.set_mid_top_right();
188  else if (v_dec < 0.75)
189  bitmap.set_mid_bottom_right();
190  else
191  bitmap.set_bottom_right();
192  }
193  }
