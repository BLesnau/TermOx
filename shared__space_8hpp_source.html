69  template <typename Children_span>
70  void disperse(Children_span& children, int surplus)
71  {
72  auto given_away = -1;
73  while (given_away != 0) {
74  given_away = 0;
75  for (auto iter = children.begin_max(); iter != children.end();
76  ++iter) {
77  auto const& policy = iter.get_policy();
78  if (policy.is_passive())
79  continue;
80  auto const max_length = policy.max();
81  auto const stretch_ratio =
82  policy.stretch() / children.total_stretch();
83  auto to_add = static_cast<std::size_t>(stretch_ratio * surplus);
84  if ((iter->length + to_add) > max_length)
85  to_add = max_length - iter->length;
86  iter->length += to_add;
87  given_away += to_add;
88  }
89  surplus -= given_away;
